- $dp_{i, j} =1..i$ 排列有 $j$ 組逆敘述對的方法數
- $dp_{i,j}=\sum_{k=0..n-1} dp_{i-1,j-k}$
- 前綴優化 $dp_{i,j} = dp_{i - 1,j} + dp_{i,j - 1}$

```cpp
#include <bits/stdc++.h>
#define int long long
#define pii pair<int, int>
#define pb push_back
#define mk make_pair
#define F first
#define S second
#define ALL(x) x.begin(), x.end()

using namespace std;
 
const int INF = 2e18;
const int maxn = 505;
const int M = 1e9 + 7;

int n, K;
int dp[maxn][maxn * maxn];

void init() {
    cin >> n >> K;
}

void solve () {
    dp[1][0] = 1; // 只有 1 個東西
    int sum, p;
    for(int i = 2; i <= n; i++){
        sum = p = 0; 
        // 前綴優化
        // 目前的總和 sum
        // 底線 p
        for(int j = 0; j <= K; j++){
            // dp[i][j] = dp[i - 1][j] + dp[i][j - 1]
            if(j - p > i - 1) { // 超過底線
                sum -= dp[i - 1][p];
                p++;
            }
            sum += dp[i - 1][j];
            dp[i][j] = sum % M;
        }
    }

    cout << dp[n][K] << "\n";
} 
 
signed main() {
    // ios::sync_with_stdio(0);
    // cin.tie(0);
    int t = 1;
    //cin >> t;
    while (t--) {
        init();
        solve();
    }
}
```
